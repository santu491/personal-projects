/**
 * A typical api function: takes an arbitrary number of arguments of type A
 * and returns a Promise which resolves with a specific response type of R.
 */
export type ApiFn<R, A extends unknown[] = []> = (...args: A) => Promise<R>;
 
/**
 * An updater function: has a similar signature with the original api function,
 * but doesn't return anything because it only triggers new api calls.
 */
export type UpdaterFn<A extends unknown[] = []> = (...args: A) => void;
 
/**
 * A simple data reader function: returns the response type R.
 */
type DataFn<R> = () => R;
/**
 * A lazy data reader function: can return the response type R or `undefined`.
 */
type LazyDataFn<R> = () => R | undefined;

export function resourceCache<R, A extends unknown[]>(
  apiFn?: ApiFn<R, A>
): {
  clear: () => void;
  delete: (...params: A | never[]) => boolean;
  get: (...params: A | never[]) => DataFn<R> | undefined;
  set: (dataFn: DataFn<R>, ...params: A | never[]) => Map<string, DataFn<R>>;
};
export function resourceCache(): {
  clear: () => void;
};
export function resourceCache<R, A extends unknown[]>(apiFn?: ApiFn<R, A>) {
  if (!apiFn) {
    return {
      clear: () => (caches = new WeakMap()),
    };
  }
 
  // initialize a new cache for this api function if it doesn't exist
  if (!caches.has(apiFn)) {
    caches.set(apiFn, new Map());
  }
 
  // get the cache for this api function
  const apiCache: Map<string, DataFn<R>> = caches.get(apiFn);
 
  // return an object with helper methods to manage the cache for this api function
  return {
    // gets the cached data reader for the given params
    get(...params: A | never[]) {
      return apiCache.get(hash(params));
    },
    // caches the data reader for the given params
    set(dataFn: DataFn<R>, ...params: A | never[]) {
      return apiCache.set(hash(params), dataFn);
    },
    // deletes the cached data reader for the given params
    delete(...params: A | never[]) {
      return apiCache.delete(hash(params));
    },
    // clears the entire cache for this api function
    clear() {
      caches.delete(apiFn);
      apiCache.clear();
    },
  };
}
 
const UNRESOLVED = Symbol();
type Unresolved = typeof UNRESOLVED;
 
function executeAndWrapPromise<Data, ArgTypes extends unknown[]>(
  apiFn: ApiFn<Data, ArgTypes>,
  ...parameters: ArgTypes
): DataFn<Data> {
  let result: Data | Unresolved = UNRESOLVED;
  let error: Error | undefined;
  const suspender = apiFn(...parameters).then(
    (r) => {
      result = r;
    },
    (e) => {
      error = e;
      // delete cache in case of error after short time to allow retry
      setTimeout(() => resourceCache(apiFn).delete(...parameters), 1000);
    }
  );
  return () => {
    if (result !== UNRESOLVED) {
      return result;
    }
    if (error) {
      throw error;
    }
    // eslint-disable-next-line @typescript-eslint/no-throw-literal
    throw suspender;
  };
}

/**
 * Wrapper for an apiFunction with params.
 * It takes the api function and all its expected arguments.
 * Also returns a data reader.
 *
 * @param apiFn A typical api function with parameters.
 * @param parameters An arbitrary number of parameters.
 */
function initializeDataReader<ResponseType, ArgTypes extends unknown[]>(

// implementation that covers the above overloads
function initializeDataReader<ResponseType, ArgTypes extends unknown[] = []>(
  apiFn: ApiFn<ResponseType, ArgTypes>,
  ...parameters: ArgTypes
): DataFn<ResponseType> {
  const apiFnCache = resourceCache(apiFn);
  const cachedResource = apiFnCache.get(...parameters);
 
  if (cachedResource !== undefined) {
    return cachedResource;
  }
 
  const dataReader = executeAndWrapPromise(apiFn, ...parameters);
 
  apiFnCache.set(dataReader, ...parameters);
 
  return dataReader;
}

export function useAsyncResource<ResponseType, ArgTypes extends unknown[]>(
  apiFunction: ApiFn<ResponseType> | ApiFn<ResponseType, ArgTypes>,
  ...parameters: ArgTypes
) {
  // keep the data reader inside a mutable object ref
  // always initialize with a lazy data reader, as it can be overwritten by the useMemo immediately
  const dataReaderObj = useRef<DataFn<ResponseType> | LazyDataFn<ResponseType>>(() => undefined);
 
  const paramsHash = hash(parameters);
  // like useEffect, but runs immediately
  useMemo(() => {
    if (parameters.length) {
      // eager initialization for api functions that don't accept arguments
      dataReaderObj.current =
        !apiFunction.length &&
        // but the user passed an empty array as the only parameter
        parameters.length === 1 &&
        Array.isArray(parameters[0]) &&
        parameters[0].length === 0
          ? initializeDataReader(apiFunction as ApiFn<ResponseType>)
          : initializeDataReader(apiFunction as ApiFn<ResponseType, ArgTypes>, ...parameters);
    }
    // eslint-disable-next-line react-hooks/exhaustive-deps
  }, [apiFunction, paramsHash]);
 
  // state to force re-render
  const [, forceRender] = useState(0);
 
  const updaterFn = useCallback(
    (...newParameters: ArgTypes) => {
      // clear any cached thing?
 
      // update the object ref
      dataReaderObj.current = initializeDataReader(apiFunction as ApiFn<ResponseType, ArgTypes>, ...newParameters);
      // update state to force a re-render
      forceRender((ct) => 1 - ct);
    },
    [apiFunction]
  );
 
  return [dataReaderObj.current, updaterFn];
}
 
const createDummyApiFn =
  <T>(defaultResponse: T | boolean) =>
  // we can't use an async function or a real promise here, as the success case needs to be sync to prevent suspending
  (): Promise<T> => {
    return {
      then: (onSuccess, onError) => {
        if (typeof defaultResponse === 'boolean') {
          const dummyError = new Error(
            'dummyApiFn error - if you’re seeing this, a dummy data reader function was called unexpectedly'
          );
          // onError needs to be async
          setImmediate(() => onError?.(dummyError));
          return;
        }
        // onSuccess needs to be sync to prevent suspending
        onSuccess?.(defaultResponse);
      },
    } as Promise<T>;
  };

export function useAsyncResourceWhen<ResponseType, ArgTypes extends unknown[]>(
  whenOrDefaultResponse: boolean | ResponseType,
  apiFunction: ApiFn<ResponseType> | ApiFn<ResponseType, ArgTypes>,
  ...parameters: ArgTypes
) {
  // we can useMemo here as the component shouldn't ever suspend in the dummy case
  const dummyApiFn = useMemo(() => createDummyApiFn(whenOrDefaultResponse), [whenOrDefaultResponse]);
 
  return useAsyncResource(whenOrDefaultResponse === true ? apiFunction : dummyApiFn, ...parameters);
}
 
export async function waitForAsyncResource<Data>(reader: DataFn<Data>): Promise<Data> {
  try {
    return reader();
  } catch (suspender) {
    if (suspender instanceof Promise) {
      await suspender;
      return reader();
    }
    throw suspender;
  }
